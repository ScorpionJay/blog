基本用法

git init 	git初始化
git remote

git add			添加修改
git commit -"message"	提交修改
git push		推送到远程

git status		查看状态
git checkout		下载
git branch		分支
git log			记录
git diff		检查不同
git merge		合并
git branch -b dev	新增分支dev并切换到dev
git branch -d dev	删除分支dev
